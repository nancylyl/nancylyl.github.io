关于Js性能2--非空整数数组加1

需求：给定一个由整数组成的非空数组所表示的非负整数，在该数的基础上加一。
最高位数字存放在数组的首位， 数组中每个元素只存储单个数字。
你可以假设除了整数 0 之外，这个整数不会以零开头。
输入: [1,2,3]
输出: [1,2,4]
解释: 输入数组表示数字 123。

在es10之前，我们没有BigInt，当我们在处理大数相加 一般会采取第三方插件或者是自己利用末尾进位相加法


—javascript
//方法1
var plusOne = function(digits) {

// 数值6145390195186705544超出Number基本类型的容纳范围，改用BigInt基本类型
let num = BigInt(digits.join(''));
// BigInt基本类型进行数学操作时，需要在数字字面量后加个n
let string = String(num + 1n);
let ary = string.split('');

return ary.map(str => Number(str));

};

var digits = [6, 1, 4, 5, 3, 9, 0, 1, 9, 5, 1, 8, 6, 7, 0, 5, 5, 4, 4];
console.log(plusOne(digits));

//方法2
const plusArray = (digits) => {
let len = digits.length;
let stack = [];
let newArr = [];

for (let i = len - 1; i >= 0; i--) {

    stack.push(digits[i]); // 元素进栈

    if (stack.length == 1) { // 栈底元素需要进位时 

        if (stack[0] == 9) { // 栈底元素为9
            if (i == 0) { //  元素全为9的情况
                newArr.unshift(0);
                stack[0] = 1; // 手动进位
            } else {
                newArr.unshift(0); // 元素不全为9 的情况
                stack = []; // 清空栈
            }
        } else {
            stack[0] = stack[0] + 1; // 栈底不用进位，栈底直接加一 （0-8）           
        }

    }
}
let endArr = stack.reverse().concat(newArr);
return endArr;

}

console.log(plusArray([1, 2, 4, 5, 9, 9]))

//方法3
function sumBigNumber(a, b) {
var res = ‘’,
temp = 0;
a = a.split(‘’);
b = b.split(‘’);
while (a.length || b.length || temp) {
temp += a.pop() + ~b.pop();//~~的作用是去掉小bai数部分，因为位运算的操作值要求是整数，其du结果也是整数，所以经过位运算的都会自动变成整数。
res = (temp % 10) + res;
temp = temp > 9;
}
return res.replace(/^0+/, ‘’);
}

var plusOne = function(digits) {
digits = digits.join(‘’);
digits = […(sumBigNumber(digits, ‘1’) + ‘’)].map(item => parseInt(item, 10));
return digits;
};

var digits = [6, 1, 4, 5, 3, 9, 0, 1, 9, 5, 1, 8, 6, 7, 0, 5, 5, 4, 4];
console.log(plusOne(digits));
